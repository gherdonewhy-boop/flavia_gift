const puzzle = [
[5,3,0,0,7,0,0,0,0],

[0,9,8,0,0,0,0,6,0],

[0,6,0,0,0,0,2,8,0],
[0,0,0,4,1,9,0,0,5],
[0,0,0,0,8,0,0,7,9]
];

// Soluzione corretta
const solution = [
[5,3,4,6,7,8,9,1,2],
[6,7,2,1,9,5,3,4,8],
[1,9,8,3,4,2,5,6,7],
[8,5,9,7,6,1,4,2,3],
[4,2,6,8,5,3,7,9,1],
[7,1,3,9,2,4,8,5,6],

[2,8,7,4,1,9,6,3,5],
[3,4,5,2,8,6,1,7,9]
];

const table = document.getElementById("sudoku");

for (let r = 0; r < 9; r++) {
  const row = document.createElement("tr");
  for (let c = 0; c < 9; c++) {
    const cell = document.createElement("td");
    const input = document.createElement("input");
    input.maxLength = 1;

    if (puzzle[r][c] !== 0) {
      input.value = puzzle[r][c];
      input.disabled = true;
      input.classList.add("fixed");
    }

    input.id = `cell-${r}-${c}`;
    cell.appendChild(input);
    row.appendChild(cell);
  }
  table.appendChild(row);
}

function checkSudoku() {
  for (let r = 0; r < 9; r++) {
    for (let c = 0; c < 9; c++) {
      const val = document.getElementById(`cell-${r}-${c}`).value;
      if (val != solution[r][c]) {
        alert("âŒ Sudoku non corretto");
        return;
      }
    }
  }
  alert("ðŸŽ‰ Sudoku completato!");
  document.getElementById("nextBtn").style.display = "inline-block";
}
